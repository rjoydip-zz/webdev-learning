# Promise

```js
async function externalServiceCall(filename) {
  return new Promise((resolve, reject) => {
    const isError = false;

    if (isError) {
      reject(Error("Error occured from externalServiceCall"));
    } else {
      resolve(filename);
    }
  });
}
```
